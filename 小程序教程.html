Page({
  data: {
    num: 1,
    arr: [{ num: 1 }, { num: 1 }, { num: 1 }],
  },
  add() {

      this.setData({
        [`arr[${idx}].num`]: item.num + 1,
      });
    });
  },
});

增

Page({
  data: {
    num: 1,
    arr: [{ num: 1 }, { num: 1 }, { num: 1 }],
  },
  add() {
    this.data.arr.forEach((item, idx) => {
      this.setData({
        [`arr[${idx}].num`]: item.num + 1,
      });
    });
  },
});

删

<view wx:for="" wx:key="index">

    <button bind:tap="remove" data-idx="">删除</button>
</view>

Page({
  data: {
    num: 1,
    arr: [{ num: "张三" }, { num: "李四" }, { num: "老王" }],
  },
  remove(event) {
    let idx = event.target.dataset.idx;
    let newArr = this.data.arr;
    newArr.splice(idx, 1);
    this.setData({
      arr: newArr,
    });
  },
});

改

<view wx:for="" wx:key="index">

    <button bind:tap="update" data-idx="">修改</button>
</view>

Page({
  data: {
    num: 1,
    arr: [{ num: "张三" }, { num: "李四" }, { num: "老王" }],
  },
  update(event) {
    let idx = event.target.dataset.idx;
    this.setData({
      [`arr[${idx}]`]: { name: "修改" },
    });
  },
});

  }

//pages/index/index.wxml
<acom></acom>
<bcom></bcom>

//utils/eventBus.js
class EventBus {
  constructor() {
    this.list = [];
  }
  //发布
  $emit(eventName, options) {
    setTimeout(() => {
      this.list.forEach((item) => {
        if (item.eventName === eventName) {
          item.callBack(options);
        }
      });
    }, 0);
  }
  //订阅
  $on(eventName, callBack) {
    this.list.push({ eventName, callBack });
    console.log(this.list);
  }
}
export default new EventBus();
